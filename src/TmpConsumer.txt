// ConsumeClaim must start a consumer loop of ConsumerGroupClaim's Messages().
func (c *CLASSNAMEConsumer) ConsumeClaim(session sarama.ConsumerGroupSession, claim sarama.ConsumerGroupClaim) error {
	for m := range claim.Messages() {
		logger.Audit("CLASSNAME: value = %s, OffSet = %v, topic = %s",
			string(m.Value), m.Offset, m.Topic)
		// Validate message
		message, err := CLASSNAMEMessageValidate(m)
		if err != nil {
			logger.Err("CLASSNAME validation failed: %s", err)
			continue
		}
		// Message action
		err = CLASSNAMEAction(message)
		if err != nil {
			logger.Err("CLASSNAME action failed: %s", err)
			continue
		}
		session.MarkMessage(message, "")
	}
	return nil
}

func startCLASSNAME(Version string, topics string) {
	logger.Audit("starting CLASSNAME consumers...")
	if App.config.Kafka.Brokers == "" {
		log.Fatal("Cant start CLASSNAME consumer, No brokers defined")
	}
	if App.config.Kafka.ConsumerTopic == "" {
		log.Fatal("Cant start CLASSNAME consumer, No topic defined")
	}
	if App.config.Kafka.Version == "" {
		log.Fatal("Cant start CLASSNAME consumer, No kafka version defined")
	}

	brokers := strings.Split(App.config.Kafka.Brokers, ",")
	version, err := sarama.ParseKafkaVersion(Version)
	if err != nil {
		log.Panicf("Error parsing Kafka version: %v", err)
	}

	config := sarama.NewConfig()
	config.Version = version
	//config.Consumer.Retry.Backoff
	config.Consumer.Offsets.Initial = sarama.OffsetOldest

	consumer := CLASSNAMEConsumer{
		ready: make(chan bool, 0),
	}

	ctx, cancel := context.WithCancel(context.Background())
	client, err := sarama.NewConsumerGroup(brokers, App.config.Kafka.ConsumerGroup, config)
	if err != nil {
		log.Panicf("Error creating CLASSNAME consumer group client: %v", err)
	}

	wg := &sync.WaitGroup{}
	go func() {
		wg.Add(1)
		defer wg.Done()
		for {
			if err := client.Consume(ctx, strings.Split(topics, ","), &consumer); err != nil {
				log.Panicf("Error from CLASSNAME consumer: %v", err)
			}
			if ctx.Err() != nil {
				return
			}
			consumer.ready = make(chan bool, 0)
		}
	}()

	<-consumer.ready // Await till the consumer has been set up
	log.Println("CLASSNAME consumer up and running!...")

	sigterm := make(chan os.Signal, 1)
	signal.Notify(sigterm, syscall.SIGINT, syscall.SIGTERM)
	select {
	case <-ctx.Done():
		log.Println("terminating: context cancelled")
	case <-sigterm:
		log.Println("terminating: via signal")
	}
	cancel()
	wg.Wait()
	if err = client.Close(); err != nil {
		log.Panicf("Error closing client: %v", err)
	}
}

// Consumer represents a Sarama consumer group consumer
type CLASSNAMEConsumer struct {
	ready chan bool
}

// Setup is run at the beginning of a new session, before ConsumeClaim
func (c *CLASSNAMEConsumer) Setup(sarama.ConsumerGroupSession) error {
	close(c.ready)
	return nil
}

// Cleanup is run at the end of a session, once all ConsumeClaim goroutines have exited
func (c *CLASSNAMEConsumer) Cleanup(sarama.ConsumerGroupSession) error {
	return nil
}
